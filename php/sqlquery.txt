CREATE TABLE IF NOT EXISTS `educator`.`table_admin` ( `id` INT(6) NOT NULL AUTO_INCREMENT COMMENT 'admin unique id' , `name` VARCHAR(30) NOT NULL COMMENT 'admin name' , `username` VARCHAR(15) NOT NULL UNIQUE COMMENT 'admin unique username' , `email` VARCHAR(50) NOT NULL UNIQUE COMMENT 'admin unique email' , `password` VARCHAR(30) NOT NULL COMMENT 'admin password' , `logintime` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT 'login time' , PRIMARY KEY (`id`)) ENGINE = InnoDB
